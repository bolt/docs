 44:     /**
 45:      * Perform a PHPUnit run and output coverage results
 46:      *
 47:      * @param string $output Target PHP file to output serialized resultst to
 48:      * @param string $test   Test directory or file to run
 49:      *
 50:      * @return integer
 51:      */
 52:     public function runPhpUnitCoverage($output, $test = null)
 53:     {
 54:         // Composer bundled PHPUnit binary
 55:         $bin = realpath(__DIR__ . '/../../vendor/phpunit/phpunit/composer/bin/phpunit');
 56: 
 57:         $builder = new ProcessBuilder();
 58:         $builder->setPrefix($bin);
 59: 
 60:         // Build the process
 61:         $process = $builder
 62:             ->setArguments(
 63:                 array(
 64:                     '--configuration',
 65:                     $this->xmlfile,
 66:                     '--coverage-php',
 67:                     $output,
 68:                     $test))
 69:             ->getProcess()
 70:             ->setTimeout(0)
 71:             ->enableOutput();
 72: 
 73:         try {
 74:             // Run it
 75:             $retval = $process->run(
 76:                 function ($type, $buffer) {
 77:                     echo $buffer;
 78:                 }
 79:             );
 80:         } catch (ProcessFailedException $e) {
 81:             echo $e->getMessage();
 82:         }
 83: 
 84:         if (!$process->isSuccessful()) {
 85:             $this->output->write('<error>' . $process->getErrorOutput() . '</error>', true);
 86:         }
 87: 
 88:         return $retval;
 89:     }

 91:     /**
 92:      * Extract the data from the PHPUnit code coverage
 93:      *
 94:      * @param PHP_CodeCoverage $codeCoverage
 95:      *
 96:      * @return array
 97:      */
 98:     private function getCoverageStats(\PHP_CodeCoverage $codeCoverage)
 99:     {
100:         $report = $codeCoverage->getReport();
101:         $classCoverage = array();
102: 
103:         foreach ($report as $item) {
104:             if (!$item instanceof \PHP_CodeCoverage_Report_Node_File) {
105:                 continue;
106:             }
107: 
108:             $classes = $item->getClassesAndTraits();
109: 
110:             foreach ($classes as $className => $class) {
111:                 $classStatements        = 0;
112:                 $coveredClassStatements = 0;
113:                 $coveredMethods         = 0;
114:                 $classMethods           = 0;
115: 
116:                 foreach ($class['methods'] as $method) {
117:                     if ($method['executableLines'] == 0) {
118:                         continue;
119:                     }
120: 
121:                     $classMethods++;
122:                     $classStatements        += $method['executableLines'];
123:                     $coveredClassStatements += $method['executedLines'];
124:                     if ($method['coverage'] == 100) {
125:                         $coveredMethods++;
126:                     }
127:                 }
128: 
129:                 if (!empty($class['package']['namespace'])) {
130:                     $namespace = '\\' . $class['package']['namespace'] . '::';
131:                 } elseif (!empty($class['package']['fullPackage'])) {
132:                     $namespace = '@' . $class['package']['fullPackage'] . '::';
133:                 } else {
134:                     $namespace = '';
135:                 }
136: 
137:                 if ($coveredClassStatements != 0) {
138:                     $classCoverage[$namespace . $className] = array(
139:                         'namespace'         => $namespace,
140:                         'className '        => $className,
141:                         'methodsCovered'    => $coveredMethods,
142:                         'methodCount'       => $classMethods,
143:                         'statementsCovered' => $coveredClassStatements,
144:                         'statementCount'    => $classStatements,
145:                     );
146:                 }
147:             }
148:         }
149: 
150:         return array(
151:             'classes'  => array(
152:                 'total'  => $report->getNumClassesAndTraits(),
153:                 'tested' => $report->getNumTestedClassesAndTraits()
154:             ),
155:             'methods'  => array(
156:                 'total'  => $report->getNumMethods(),
157:                 'tested' => $report->getNumTestedMethods()
158:             ),
159:             'lines'    => array(
160:                 'total'  => $report->getNumExecutableLines(),
161:                 'tested' => $report->getNumExecutedLines()
162:             ),
163:             'coverage' => $classCoverage
164:         );
165:     }
166: 
167:     /**
168:      * Compare two sets of results and return the increased stats
169:      *
170:      * @param string $beforeFile
171:      * @param string $afterFile
172:      *
173:      * @return array
174:      */
175:     public function compareCoverageStats($beforeFile, $afterFile)
176:     {
177:         // Before
178:         $codeCoverage = unserialize(file_get_contents($beforeFile));
179:         $before = $this->getCoverageStats($codeCoverage);
180: 
181:         // After
182:         $codeCoverage = unserialize(file_get_contents($afterFile));
183:         $after = $this->getCoverageStats($codeCoverage);
184: 
185:         $increase = array(
186:             'classes'  => 0,
187:             'methods'  => 0,
188:             'lines'    => 0,
189:             'coverage' => array()
190:         );
191: 
192:         $deltaBefore = $before['classes']['total'] - $before['classes']['tested'];
193:         $deltaAfter = $after['classes']['total'] - $after['classes']['tested'];
194:         if ($deltaAfter > $deltaBefore) {
195:             $increase['classes'] = $deltaAfter - $deltaBefore;
196:         }
197: 
198:         $deltaBefore = $before['methods']['total'] - $before['methods']['tested'];
199:         $deltaAfter = $after['methods']['total'] - $after['methods']['tested'];
200:         if ($deltaAfter > $deltaBefore) {
201:             $increase['methods'] = $deltaAfter - $deltaBefore;
202:         }
203: 
204:         $deltaBefore = $before['lines']['total'] - $before['lines']['tested'];
205:         $deltaAfter = $after['lines']['total'] - $after['lines']['tested'];
206:         if ($deltaAfter > $deltaBefore) {
207:             $increase['lines'] = $deltaAfter - $deltaBefore;
208:         }
209: 
210:         foreach ($after['coverage'] as $class => $data) {
211:             // Methods
212:             $deltaBefore = $data['methodsCovered'] - $before['coverage'][$class]['methodsCovered'];
213:             $deltaAfter = $data['methodCount'] - $before['coverage'][$class]['methodCount'];
214:             if ($deltaAfter > $deltaBefore) {
215:                 $increase['coverage'][$class]['methods'] = $deltaAfter - $deltaBefore;
216:             }
217: 
218:             // Statments
219:             $deltaBefore = $data['statementsCovered'] - $before['coverage'][$class]['statementsCovered'];
220:             $deltaAfter = $data['statementCount'] - $before['coverage'][$class]['statementCount'];
221:             if ($deltaAfter > $deltaBefore) {
222:                 $increase['coverage'][$class]['statements'] = $deltaAfter - $deltaBefore;
223:             }
224:         }
225: 
226:         return $increase;
227:     }

253:

255: 
256:         foreach ($stats['coverage'] as $class => $data) {

259: 
260: EOF;
261:         }
262: 
263:         return $coverage;
