 484:         }
 485:

 487:         // @todo use $this->setRelation() for this
 488:         if (!empty($values['relation'])) {

 490:             unset($values['relation']);
 491:         } else {
 492:             $this->relation = array();
 493:         }

 495:         // @todo check for allowed file types.

 522:                     $this->app['logger.system']->error($message, array('event' => 'upload'));

 532:                 // Check if we don't have doubles.

 546:                     $this->app['logger.system']->error("Upload: couldn't write upload '$basename'.", array('event' => 'upload'));
 547:                 }
 548:             }
 549:         }
 550:

 552:     }
