 624:     /**
 625:      * Search through a single contenttype.
 626:      *
 627:      * Search, weigh and return the results.
 628:      *
 629:      * @param       $query
 630:      * @param       $contenttype
 631:      * @param       $fields
 632:      * @param array $filter
 633:      *
 634:      * @return \Bolt\Content
 635:      */
 636:     private function searchSingleContentType($query, $contenttype, $fields, array $filter = null)
 637:     {
 638:         // This could be even more configurable
 639:         // (see also Content->getFieldWeights)
 640:         $searchableTypes = array('text', 'textarea', 'html', 'markdown');
 641:         $table = $this->getContenttypeTablename($contenttype);
 642: 
 643:         // Build fields 'WHERE'
 644:         $fieldsWhere = array();
 645:         foreach ($fields as $field => $fieldconfig) {
 646:             if (in_array($fieldconfig['type'], $searchableTypes)) {
 647:                 foreach ($query['words'] as $word) {
 648:                     $fieldsWhere[] = sprintf('%s.%s LIKE %s', $table, $field, $this->app['db']->quote('%' . $word . '%'));
 649:                 }
 650:             }
 651:         }
 652: 
 653:         // make taxonomies work
 654:         $taxonomytable = $this->getTablename('taxonomy');
 655:         $taxonomies    = $this->getContentTypeTaxonomy($contenttype);
 656:         $tagsWhere     = array();
 657:         $tagsQuery     = '';
 658:         foreach ($taxonomies as $taxonomy) {
 659:             if ($taxonomy['behaves_like'] == 'tags') {
 660:                 foreach ($query['words'] as $word) {
 661:                     $tagsWhere[] = sprintf('%s.slug LIKE %s', $taxonomytable, $this->app['db']->quote('%' . $word . '%'));
 662:                 }
 663:             }
 664:         }
 665:         // only add taxonomies if they exist
 666:         if (!empty($taxonomies) && !empty($tagsWhere)) {
 667:             $tagsQueryA = sprintf("%s.contenttype = '%s'", $taxonomytable, $contenttype);
 668:             $tagsQueryB = implode(' OR ', $tagsWhere);
 669:             $tagsQuery   = sprintf(' OR (%s AND (%s))', $tagsQueryA, $tagsQueryB);
 670:         }
 671: 
 672:         // Build filter 'WHERE"
 673:         // @todo make relations work as well
 674:         $filterWhere = array();
 675:         if (!is_null($filter)) {
 676:             foreach ($fields as $field => $fieldconfig) {
 677:                 if (isset($filter[$field])) {
 678:                     $filterWhere[] = $this->parseWhereParameter($table . '.' . $field, $filter[$field]);
 679:                 }
 680:             }
 681:         }
 682: 
 683:         // Build actual where
 684:         $where = array();
 685:         $where[] = sprintf("%s.status = 'published'", $table);
 686:         $where[] = '(( ' . implode(' OR ', $fieldsWhere) . ' ) ' . $tagsQuery . ' )';
 687:         $where = array_merge($where, $filterWhere);
 688: 
 689:         // Build SQL query
 690:         $select = sprintf(
 691:             'SELECT %s.id FROM %s LEFT JOIN %s ON %s.id = %s.content_id WHERE %s GROUP BY %s.id',
 692:             $table,
 693:             $table,
 694:             $taxonomytable,
 695:             $table,
 696:             $taxonomytable,
 697:             implode(' AND ', $where),
 698:             $table
 699:         );
 700: 
 701:         // Run Query
 702:         $results = $this->app['db']->fetchAll($select);
 703: 
 704:         if (!empty($results)) {
 705:             $ids = implode(' || ', util::array_pluck($results, 'id'));
 706: 
 707:             $results = $this->getContent($contenttype, array('id' => $ids, 'returnsingle' => false));
 708: 
 709:             // Convert and weight
 710:             foreach ($results as $result) {
 711:                 $result->weighSearchResult($query);
 712:             }
 713:         }
 714: 
 715:         return $results;
 716:     }

 718:     /**
 719:      * Compare by search weights.
 720:      *
 721:      * Or fallback to dates or title
 722:      *
 723:      * @param \Bolt\Content $a
 724:      * @param \Bolt\Content $b
 725:      *
 726:      * @return int
 727:      */
 728:     private function compareSearchWeights(Content $a, Content $b)
 729:     {
 730:         if ($a->getSearchResultWeight() > $b->getSearchResultWeight()) {
 731:             return -1;
 732:         }
 733:         if ($a->getSearchResultWeight() < $b->getSearchResultWeight()) {
 734:             return 1;
 735:         }
 736:         if ($a['datepublish'] > $b['datepublish']) {
 737:             // later is more important
 738:             return -1;
 739:         }
 740:         if ($a['datepublish'] < $b['datepublish']) {
 741:             // earlier is less important
 742:             return 1;
 743:         }
 744: 
 745:         return strcasecmp($a['title'], $b['title']);
 746:     }

 748:     /**
 749:      * Search through actual content.
 750:      *
 751:      * Unless the query is invalid it will always return a 'result array'. It may
 752:      * complain in the log but it won't abort.
 753:      *
 754:      * @param string  $q            Search string
 755:      * @param array   $contenttypes Contenttype names to search for:
 756:      *                              - string: Specific contenttype
 757:      *                              - null:   Every searchable contenttype
 758:      * @param array   $filters      Additional filters for contenttypes
 759:      *                              - key is contenttype
 760:      *                              - value is filter
 761:      * @param integer $limit        limit the number of results
 762:      * @param integer $offset       skip this number of results
 763:      *
 764:      * @return mixed false if query is invalid, an array with results if query was executed
 765:      */
 766:     public function searchContent($q, array $contenttypes = null, array $filters = null, $limit = 9999, $offset = 0)
 767:     {
 768:         $query = $this->decodeSearchQuery($q);
 769:         if (!$query['valid']) {
 770:             return false;
 771:         }
 772: 
 773:         $appCt = $this->app['config']->get('contenttypes');
 774: 
 775:         // By default we only search through searchable contenttypes
 776:         if (is_null($contenttypes)) {
 777:             $contenttypes = array_keys($appCt);
 778:             $contenttypes = array_filter(
 779:                 $contenttypes,
 780:                 function ($ct) use ($appCt) {
 781:                     if ((isset($appCt[$ct]['searchable']) && $appCt[$ct]['searchable'] === false) ||
 782:                         (isset($appCt[$ct]['viewless']) && $appCt[$ct]['viewless'] === true)
 783:                     ) {
 784:                         return false;
 785:                     }
 786: 
 787:                     return true;
 788:                 }
 789:             );
 790:             $contenttypes = array_map(
 791:                 function ($ct) use ($appCt) {
 792:                     return $appCt[$ct]['slug'];
 793:                 },
 794:                 $contenttypes
 795:             );
 796:         }
 797: 
 798:         // Build our search results array
 799:         $results = array();
 800:         foreach ($contenttypes as $contenttype) {
 801:             $ctconfig = $this->getContentType($contenttype);
 802: 
 803:             $fields = $ctconfig['fields'];
 804:             $filter = null;
 805: 
 806:             if (is_array($filters) && isset($filters[$contenttype])) {
 807:                 $filter = $filters[$contenttype];
 808:             }
 809: 
 810:             $subResults = $this->searchSingleContentType($query, $contenttype, $fields, $filter);
 811: 
 812:             $results = array_merge($results, $subResults);
 813:         }
 814: 
 815:         // Sort the results
 816:         usort($results, array($this, 'compareSearchWeights'));
 817: 
 818:         $noOfResults = count($results);
 819: 
 820:         $pageResults = array();
 821:         if ($offset < $noOfResults) {
 822:             $pageResults = array_slice($results, $offset, $limit);
 823:         }
 824: 
 825:         return array(
 826:             'query'         => $query,
 827:             'no_of_results' => $noOfResults,
 828:             'results'       => $pageResults
 829:         );
 830:     }

1423:                  * @see #1799 https://github.com/bolt/bolt/issues/1799

1950:     /**
1951:      * Helper function for sorting Records of content that have a Grouping.
1952:      *
1953:      * @param Content $a
1954:      * @param Content $b
1955:      *
1956:      * @return int
1957:      */
1958:     private function groupingSort(Content $a, Content $b)
1959:     {
1960:         // Same group, sort within group.
1961:         if ($a->group['slug'] == $b->group['slug']) {
1962:             if (!empty($a->sortorder) || !empty($b->sortorder)) {
1963:                 if (!isset($a->sortorder)) {
1964:                     return 1;
1965:                 } elseif (!isset($b->sortorder)) {
1966:                     return -1;
1967:                 } else {
1968:                     return ($a->sortorder < $b->sortorder) ? -1 : 1;
1969:                 }
1970:             }
1971: 
1972:             if (!empty($a->contenttype['sort'])) {
1973:                 // Same group, so we sort on contenttype['sort']
1974:                 list($secondSort, $order) = $this->getSortOrder($a->contenttype['sort']);
1975: 
1976:                 $vala = strtolower($a->values[$secondSort]);
1977:                 $valb = strtolower($b->values[$secondSort]);
1978: 
1979:                 if ($vala == $valb) {
1980:                     return 0;
1981:                 } else {
1982:                     $result = ($vala < $valb) ? -1 : 1;
1983:                     // if $order is false, the 'getSortOrder' indicated that we used something like '-id'.
1984:                     // So, perhaps we need to inverse the result.
1985:                     return $order ? $result : -$result;
1986:                 }
1987:             }
1988:         }
1989:         // Otherwise, sort based on the group. Or, more specifically, on the index of
1990:         // the item in the group's taxonomy definition.

1994:     /**

2000:      * @param mixed  $fieldtype
2001:      *

2003:      */

2005:     {

2009:         if (strpos($value, " || ") !== false) {
2010:             $values = explode(" || ", $value);
2011:             foreach ($values as $index => $value) {
2012:                 $values[$index] = $this->parseWhereParameter($key, $value, $fieldtype);

2021:

2026:         $operator = "=";
2027:

2029: 
2030:         if ($first == "!") {
2031:             $operator = "!=";
2032:             $value = substr($value, 1);
2033:         } elseif (substr($value, 0, 2) == "<=") {
2034:             $operator = "<=";
2035:             $value = substr($value, 2);
2036:         } elseif (substr($value, 0, 2) == ">=") {
2037:             $operator = ">=";
2038:             $value = substr($value, 2);
2039:         } elseif ($first == "<") {
