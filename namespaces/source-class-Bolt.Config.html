 384:     /**
 385:      * Parse a single Contenttype configuration array.
 386:      *
 387:      * @param string $key
 388:      * @param array  $contentType
 389:      * @param array  $generalConfig
 390:      *
 391:      * @throws LowlevelException
 392:      *
 393:      * @return array
 394:      */
 395:     protected function parseContentType($key, $contentType, $generalConfig)
 396:     {
 397:         // If the slug isn't set, and the 'key' isn't numeric, use that as the slug.
 398:         if (!isset($contentType['slug']) && !is_numeric($key)) {
 399:             $contentType['slug'] = Slugify::create()->slugify($key);
 400:         }
 401: 
 402:         // If neither 'name' nor 'slug' is set, we need to warn the user. Same goes for when
 403:         // neither 'singular_name' nor 'singular_slug' is set.
 404:         if (!isset($contentType['name']) && !isset($contentType['slug'])) {
 405:             $error = sprintf("In contenttype <code>%s</code>, neither 'name' nor 'slug' is set. Please edit <code>contenttypes.yml</code>, and correct this.", $key);
 406:             throw new LowlevelException($error);
 407:         }
 408:         if (!isset($contentType['singular_name']) && !isset($contentType['singular_slug'])) {
 409:             $error = sprintf("In contenttype <code>%s</code>, neither 'singular_name' nor 'singular_slug' is set. Please edit <code>contenttypes.yml</code>, and correct this.", $key);
 410:             throw new LowlevelException($error);
 411:         }
 412: 
 413:         if (!isset($contentType['slug'])) {
 414:             $contentType['slug'] = Slugify::create()->slugify($contentType['name']);
 415:         }
 416:         if (!isset($contentType['singular_slug'])) {
 417:             $contentType['singular_slug'] = Slugify::create()->slugify($contentType['singular_name']);
 418:         }
 419:         if (!isset($contentType['show_on_dashboard'])) {
 420:             $contentType['show_on_dashboard'] = true;
 421:         }
 422:         if (!isset($contentType['show_in_menu'])) {
 423:             $contentType['show_in_menu'] = true;
 424:         }
 425:         if (!isset($contentType['sort'])) {
 426:             $contentType['sort'] = false;
 427:         }
 428:         if (!isset($contentType['default_status'])) {
 429:             $contentType['default_status'] = 'draft';
 430:         }
 431:         if (!isset($contentType['viewless'])) {
 432:             $contentType['viewless'] = false;
 433:         }
 434:         if (!isset($contentType['liveeditor'])) {
 435:             $contentType['liveeditor'] = true;
 436:         }
 437:         // Override contenttype setting with view and config settings
 438:         if (($contentType['viewless']) || (!$generalConfig['liveeditor'])) {
 439:             $contentType['liveeditor'] = false;
 440:         }
 441:         // Allow explicit setting of a Contenttype's table name suffix. We default
 442:         // to slug if not present as it has been this way since Bolt v1.2.1
 443:         if (!isset($contentType['tablename'])) {
 444:             $contentType['tablename'] = $contentType['slug'];
 445:         } else {
 446:             $contentType['tablename'] = Slugify::create()->slugify($contentType['tablename']);
 447:         }
 448: 
 449:         list($fields, $groups) = $this->parseFieldsAndGroups($contentType['fields'], $generalConfig);
 450:         $contentType['fields'] = $fields;
 451:         $contentType['groups'] = $groups;
 452: 
 453:         // Make sure taxonomy is an array.
 454:         if (isset($contentType['taxonomy']) && !is_array($contentType['taxonomy'])) {
 455:             $contentType['taxonomy'] = array($contentType['taxonomy']);
 456:         }
 457: 
 458:         // when adding relations, make sure they're added by their slug. Not their 'name' or 'singular name'.
 459:         if (!empty($contentType['relations']) && is_array($contentType['relations'])) {
 460:             foreach (array_keys($contentType['relations']) as $relkey) {
 461:                 if ($relkey != Slugify::create()->slugify($relkey)) {
 462:                     $contentType['relations'][Slugify::create()->slugify($relkey)] = $contentType['relations'][$relkey];
 463:                     unset($contentType['relations'][$relkey]);
 464:                 }
 465:             }
 466:         }
 467: 
 468:         return $contentType;
 469:     }

 471:     /**
 472:      * Parse a Contenttype's filed and determine the grouping
 473:      *
 474:      * @param array $fields
 475:      * @param array $generalConfig
 476:      *
 477:      * @return array
 478:      */
 479:     protected function parseFieldsAndGroups(array $fields, array $generalConfig)
 480:     {
 481:         $acceptableFileTypes = $generalConfig['accept_file_types'];
 482: 
 483:         $currentGroup = 'ungrouped';
 484:         $groups = array();
 485:         $hasGroups = false;
 486: 
 487:         foreach ($fields as $key => $field) {
 488:             unset($fields[$key]);
 489:             $key = str_replace('-', '_', strtolower(Str::makeSafe($key, true)));
 490: 
 491:             // If field is a "file" type, make sure the 'extensions' are set, and it's an array.
 492:             if ($field['type'] == 'file' || $field['type'] == 'filelist') {
 493:                 if (empty($field['extensions'])) {
 494:                     $field['extensions'] = $acceptableFileTypes;
 495:                 }
 496: 
 497:                 if (!is_array($field['extensions'])) {
 498:                     $field['extensions'] = array($field['extensions']);
 499:                 }
 500:             }
 501: 
 502:             // If field is an "image" type, make sure the 'extensions' are set, and it's an array.
 503:             if ($field['type'] == 'image' || $field['type'] == 'imagelist') {
 504:                 if (empty($field['extensions'])) {
 505:                     $field['extensions'] = array_intersect(
 506:                         array('gif', 'jpg', 'jpeg', 'png'),
 507:                         $acceptableFileTypes
 508:                     );
 509:                 }
 510: 
 511:                 if (!is_array($field['extensions'])) {
 512:                     $field['extensions'] = array($field['extensions']);
 513:                 }
 514:             }
 515: 
 516:             // If field is a "Select" type, make sure the array is a "hash" (as opposed to a "map")
 517:             // For example: [ 'yes', 'no' ] => { 'yes': 'yes', 'no': 'no' }
 518:             // The reason that we do this, is because if you set values to ['blue', 'green'], that is
 519:             // what you'd expect to see in the database. Not '0' and '1', which is what would happen,
 520:             // if we didn't "correct" it here.
 521:             // @see used hack: http://stackoverflow.com/questions/173400/how-to-check-if-php-array-is-associative-or-sequential
 522:             if ($field['type'] == 'select' && isset($field['values']) && is_array($field['values']) &&
 523:                 array_values($field['values']) === $field['values']) {
 524:                 $field['values'] = array_combine($field['values'], $field['values']);
 525:             }
 526: 
 527:             if (!empty($field['group'])) {
 528:                 $hasGroups = true;
 529:             }
 530: 
 531:             // Make sure we have these keys and every field has a group set
 532:             $field = array_replace(
 533:                 array(
 534:                     'label'   => '',
 535:                     'variant' => '',
 536:                     'default' => '',
 537:                     'pattern' => '',
 538:                     'group'   => $currentGroup,
 539:                 ),
 540:                 $field
 541:             );
 542: 
 543:             // Collect group data for rendering.
 544:             // Make sure that once you started with group all following have that group, too.
 545:             $currentGroup = $field['group'];
 546:             $groups[$currentGroup] = 1;
 547: 
 548:             // Prefix class with "form-control"
 549:             $field['class'] = 'form-control' . (isset($field['class']) ? ' ' . $field['class'] : '');
 550: 
 551:             $fields[$key] = $field;
 552:         }
 553: 
 554:         // Make sure the 'uses' of the slug is an array.
 555:         if (isset($fields['slug']) && isset($fields['slug']['uses']) &&
 556:             !is_array($fields['slug']['uses'])
 557:         ) {
 558:             $fields['slug']['uses'] = array($fields['slug']['uses']);
 559:         }
 560: 
 561:         return array($fields, $hasGroups ? array_keys($groups) : false);
 562:     }

 564:     /**
 565:      * Parse and fine-tune the database configuration.
 566:      *
 567:      * @param array $options
 568:      *
 569:      * @return array
 570:      */
 571:     protected function parseDatabase(array $options)
 572:     {
 573:         // Make sure prefix ends with underscore
 574:         if (substr($options['prefix'], strlen($options['prefix']) - 1) !== '_') {
 575:             $options['prefix'] .= '_';
 576:         }
 577: 
 578:         // Parse master connection parameters
 579:         $master = $this->parseConnectionParams($options);
 580:         // Merge master connection into options
 581:         $options = array_replace($options, $master);
 582: 
 583:         // Add platform specific random functions
 584:         $driver = Str::replaceFirst('pdo_', '', $options['driver']);
 585:         if ($driver === 'sqlite') {
 586:             $options['driver'] = 'pdo_sqlite';
 587:             $options['randomfunction'] = 'RANDOM()';
 588:         } elseif (in_array($driver, array('mysql', 'mysqli'))) {
 589:             $options['driver'] = 'pdo_mysql';
 590:             $options['randomfunction'] = 'RAND()';
 591:         } elseif (in_array($driver, array('pgsql', 'postgres', 'postgresql'))) {
 592:             $options['driver'] = 'pdo_pgsql';
 593:             $options['randomfunction'] = 'RANDOM()';
 594:         }
 595: 
 596:         // Parse SQLite separately since it has to figure out database path
 597:         if ($driver === 'sqlite') {
 598:             return $this->parseSqliteOptions($options);
 599:         }
 600: 
 601:         // If no slaves return with single connection
 602:         if (empty($options['slaves'])) {
 603:             return $options;
 604:         }
 605: 
 606:         // Specify we want a master slave connection
 607:         $options['wrapperClass'] = '\Doctrine\DBAL\Connections\MasterSlaveConnection';
 608: 
 609:         // Add master connection where MasterSlaveConnection looks for it.
 610:         $options['master'] = $master;
 611: 
 612:         // Parse each slave connection parameters
 613:         foreach ($options['slaves'] as $name => $slave) {
 614:             $options['slaves'][$name] = $this->parseConnectionParams($slave, $master);
 615:         }
 616: 
 617:         return $options;
 618:     }

 620:     /**
 621:      * Fine-tune Sqlite configuration parameters.
 622:      *
 623:      * @param array $config
 624:      *
 625:      * @return array
 626:      */
 627:     protected function parseSqliteOptions(array $config)
 628:     {
 629:         if (isset($config['memory']) && $config['memory']) {
 630:             // If in-memory, no need to parse paths
 631:             unset($config['path']);
 632: 
 633:             return $config;
 634:         } else {
 635:             // Prevent SQLite driver from trying to use in-memory connection
 636:             unset($config['memory']);
 637:         }
 638: 
 639:         // Get path from config or use database path
 640:         if (isset($config['path'])) {
 641:             $path = $this->app['pathmanager']->create($config['path']);
 642:             // If path is relative, resolve against root path
 643:             if ($path instanceof RelativePathInterface) {
 644:                 $path = $path->resolveAgainst($this->app['resources']->getPathObject('root'));
 645:             }
 646:         } else {
 647:             $path = $this->app['resources']->getPathObject('database');
 648:         }
 649: 
 650:         // If path has filename with extension, use that
 651:         if ($path->hasExtension()) {
 652:             $config['path'] = $path->string();
 653: 
 654:             return $config;
 655:         }
 656: 
 657:         // Use database name for filename
 658:         /** @var PathInterface $filename */
 659:         $filename = $this->app['pathmanager']->create(basename($config['dbname']));
 660:         if (!$filename->hasExtension()) {
 661:             $filename = $filename->joinExtensions('db');
 662:         }
 663: 
 664:         // Join filename with database path
 665:         $config['path'] = $path->joinAtoms($filename)->string();
 666: 
 667:         return $config;
 668:     }

 670:     /**
 671:      * Parses params to valid connection parameters:
 672:      * - Defaults are merged into the params
 673:      * - Bolt keys are converted to Doctrine keys
 674:      * - Invalid keys are filtered out
 675:      *
 676:      * @param array|string $params
 677:      * @param array        $defaults
 678:      *
 679:      * @return array
 680:      */
 681:     protected function parseConnectionParams($params, $defaults = array())
 682:     {
 683:         // Handle host shortcut
 684:         if (is_string($params)) {
 685:             $params = array('host' => $params);
 686:         }
 687: 
 688:         // Convert keys from Bolt
 689:         $replacements = array(
 690:             'databasename' => 'dbname',
 691:             'username'     => 'user',
 692:         );
 693:         foreach ($replacements as $old => $new) {
 694:             if (isset($params[$old])) {
 695:                 $params[$new] = $params[$old];
 696:                 unset($params[$old]);
 697:             }
 698:         }
 699: 
 700:         // Merge in defaults
 701:         $params = array_replace($defaults, $params);
 702: 
 703:         // Filter out invalid keys
 704:         $validKeys = array(
 705:             'user', 'password', 'host', 'port', 'dbname', 'charset',      // common
 706:             'path', 'memory',                                             // Qqlite
 707:             'unix_socket', 'driverOptions',                               // MySql
 708:             'sslmode',                                                    // PostgreSQL
 709:             'servicename', 'service', 'pooled', 'instancename', 'server', // Oracle
 710:             'persistent',                                                 // SQL Anywhere
 711:         );
 712:         $params = array_intersect_key($params, array_flip($validKeys));
 713: 
 714:         return $params;
 715:     }

 717:     /**
 718:      * Sanity checks for doubles in in contenttypes.
 719:      */
 720:     public function checkConfig()
 721:     {
 722:         $slugs = array();
 723: 
 724:         $wrongctype = false;
 725: 
 726:         foreach ($this->data['contenttypes'] as $key => $ct) {
 727:             /**
 728:              * Make sure any field that has a 'uses' parameter actually points to a field that exists.
 729:              *
 730:              * For example, this will show a notice:
 731:              * entries:
 732:              *   name: Entries
 733:              *     singular_name: Entry
 734:              *     fields:
 735:              *       title:
 736:              *         type: text
 737:              *         class: large
 738:              *       slug:
 739:              *         type: slug
 740:              *         uses: name
 741:              */
 742:             foreach ($ct['fields'] as $fieldname => $field) {
 743:                 // Verify that the contenttype doesn't try to add fields that are reserved.
 744:                 if ($fieldname != 'slug' && in_array($fieldname, $this->reservedFieldNames)) {
 745:                     $error = Trans::__(
 746:                         'contenttypes.generic.reserved-name',
 747:                         array('%contenttype%' => $key, '%field%' => $fieldname)
 748:                     );
 749:                     $this->app['session']->getFlashBag()->add('error', $error);
 750: 
 751:                     return;
 752:                 }
 753: 
 754:                 // Check 'uses'. If it's an array, split it up, and check the separate parts. We also need to check
 755:                 // for the fields that are always present, like 'id'.
 756:                 if (is_array($field) && !empty($field['uses'])) {
 757:                     foreach ($field['uses'] as $useField) {
 758:                         if (!empty($field['uses']) && empty($ct['fields'][$useField]) && !in_array($useField, $this->reservedFieldNames)) {
 759:                             $error = Trans::__(
 760:                                 'contenttypes.generic.wrong-use-field',
 761:                                 array('%contenttype%' => $key, '%field%' => $fieldname, '%uses%' => $useField)
 762:                             );
 763:                             $this->app['session']->getFlashBag()->add('error', $error);
 764: 
 765:                             return;
 766:                         }
 767:                     }
 768:                 }
 769: 
 770:                 // Make sure the 'type' is in the list of allowed types
 771:                 if (!isset($field['type']) || !$this->fields->has($field['type'])) {
 772:                     $error = Trans::__(
 773:                         'contenttypes.generic.no-proper-type',
 774:                         array('%contenttype%' => $key, '%field%' => $fieldname, '%type%' =>
 775:                          $field['type'])
 776:                     );
 777:                     $this->app['session']->getFlashBag()->add('error', $error);
 778:                     $wrongctype = true && $this->app['users']->getCurrentUsername();
 779:                 }
 780:             }
 781: 
 782:             // Keep a running score of used slugs.
 783:             if (!isset($slugs[$ct['slug']])) {
 784:                 $slugs[$ct['slug']] = 0;
 785:             }
 786:             $slugs[$ct['slug']]++;
 787:             if (!isset($slugs[$ct['singular_slug']])) {
 788:                 $slugs[$ct['singular_slug']] = 0;
 789:             }
 790:             if ($ct['singular_slug'] != $ct['slug']) {
 791:                 $slugs[$ct['singular_slug']]++;
 792:             }
 793:         }
 794: 
 795:         // Check DB-tables integrity
 796:         if (!$wrongctype && $this->app['integritychecker']->needsCheck() &&
 797:            (count($this->app['integritychecker']->checkTablesIntegrity()) > 0) &&
 798:             $this->app['users']->getCurrentUsername()) {
 799:             $msg = Trans::__(
 800:                 "The database needs to be updated/repaired. Go to 'Configuration' > '<a href=\"%link%\">Check Database</a>' to do this now.",
 801:                 array('%link%' => Lib::path('dbcheck'))
 802:             );
 803:             $this->app['session']->getFlashBag()->add('error', $msg);
 804: 
 805:             return;
 806:         }
 807: 
 808:         // Sanity checks for taxonomy.yml
 809:         foreach ($this->data['taxonomy'] as $key => $taxo) {
 810:             // Show some helpful warnings if slugs or keys are not set correctly.
 811:             if ($taxo['slug'] != $key) {
 812:                 $error = Trans::__(
 813:                     "The identifier and slug for '%taxonomytype%' are the not the same ('%slug%' vs. '%taxonomytype%'). Please edit taxonomy.yml, and make them match to prevent inconsistencies between database storage and your templates.",
 814:                     array('%taxonomytype%' => $key, '%slug%' => $taxo['slug'])
 815:                 );
 816:                 $this->app['session']->getFlashBag()->add('error', $error);
 817: 
 818:                 return;
 819:             }
 820:         }
 821: 
 822:         // if there aren't any other errors, check for duplicates across contenttypes.
 823:         if (!$this->app['session']->getFlashBag()->has('error')) {
 824:             foreach ($slugs as $slug => $count) {
 825:                 if ($count > 1) {
 826:                     $error = Trans::__(
 827:                         "The slug '%slug%' is used in more than one contenttype. Please edit contenttypes.yml, and make them distinct.",
 828:                         array('%slug%' => $slug)
 829:                     );
 830:                     $this->app['session']->getFlashBag()->add('error', $error);
 831: 
 832:                     return;
 833:                 }
 834:             }
 835:         }
 836:     }
